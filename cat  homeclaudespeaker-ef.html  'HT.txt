function calcScore(sp,imp,car) { return (spMap[sp]||0)+(impMap[imp]||0)-(carMap[car]||0); }
function calcEsito(score,agenda) { if(agenda==='No') return 'Accettare con downgrade'; if(score<0) return 'Rifiutare'; return 'Accettare'; }
function calcFin(esito,sp) { if(esito==='Accettare') return sp; if(esito==='Accettare con downgrade') return dgMap[sp]||sp; return '—'; }

function esitoStyle(e) {
  if(e==='Accettare') return {bg:'var(--green-bg)',color:'var(--green)',bar:'var(--green-mid)',cls:'badge-green'};
  if(e==='Accettare con downgrade') return {bg:'var(--amber-bg)',color:'var(--amber)',bar:'var(--amber-mid)',cls:'badge-amber'};
  return {bg:'var(--red-bg)',color:'var(--red)',bar:'var(--red-mid)',cls:'badge-red'};
}

function calcola() {
  const sp=v('f-speaker'),imp=v('f-imp'),car=v('f-carico'),ag=v('f-agenda'),str=v('f-strategico');
  if(!sp||!imp||!car||!ag||!str){document.getElementById('result-area').style.display='none';return;}
  const score=calcScore(sp,imp,car), esito=calcEsito(score,ag), fin=calcFin(esito,sp);
  const s=esitoStyle(esito);
  document.getElementById('result-area').style.display='block';
  const circ=document.getElementById('r-circle');
  circ.textContent=score; circ.style.background=s.bg; circ.style.color=s.color;
  const ep=document.getElementById('r-esito');
  ep.textContent=esito; ep.style.background=s.bg; ep.style.color=s.color;
  document.getElementById('r-speaker-fin').textContent=fin;
  document.getElementById('r-score-lbl').textContent=score+' pt';
  const pct=Math.max(4,Math.min(100,((score+3)/9)*100));
  const bar=document.getElementById('r-bar'); bar.style.width=pct+'%'; bar.style.background=s.bar;
  const critData=[
    {l:'Speaker',v:sp,pts:spMap[sp],pos:true},
    {l:'Importanza',v:imp,pts:impMap[imp],pos:true},
    {l:'Carico',v:car,pts:-(carMap[car]),pos:carMap[car]<2},
    {l:'Agenda',v:ag,pts:null,pos:ag==='Sì'},
    {l:'Strategico EF',v:str,pts:null,pos:str==='Sì'},
    {l:'Score finale',v:score+' pt',pts:null,pos:score>=0},
  ];
  document.getElementById('r-criteri').innerHTML=critData.map(c=>`
    <div class="crit-item">
      <div class="crit-label">${c.l}</div>
      <div class="crit-val">${c.v}</div>
      ${c.pts!==null?`<div class="crit-pts" style="color:${c.pos?'var(--green)':'var(--red)'}">${c.pts>0?'+':''}${c.pts} pt</div>`:''}
    </div>`).join('');
}

function salva() {
  const nome=v('f-nome').trim(),sp=v('f-speaker'),imp=v('f-imp'),car=v('f-carico'),ag=v('f-agenda'),str=v('f-strategico');
  if(!nome||!sp||!imp||!car||!ag||!str){alert('Compila almeno: nome evento, speaker, importanza, carico, agenda e strategico.');return;}
  const score=calcScore(sp,imp,car),esito=calcEsito(score,ag);
  events.push({nome,data:v('f-data'),luogo:v('f-luogo'),speaker:sp,importanza:imp,carico:car,agenda:ag,strategico:str,note:v('f-note'),score,esito,speakerFinale:calcFin(esito,sp),ts:Date.now()});
  save();
  const ok=document.getElementById('save-ok'); ok.style.display='flex';
  setTimeout(()=>ok.style.display='none',3000);
}

function resetForm() {
  ['f-nome','f-data','f-luogo','f-note'].forEach(id=>document.getElementById(id).value='');
  ['f-speaker','f-imp','f-carico','f-agenda','f-strategico'].forEach(id=>document.getElementById(id).value='');
  document.getElementById('result-area').style.display='none';
}

function v(id){return document.getElementById(id).value;}

function fmtDate(d){
  if(!d)return'—';
  try{return new Date(d).toLocaleDateString('it-IT',{day:'2-digit',month:'short',year:'numeric'});}catch(e){return d;}
}

function renderDashboard(){
  const tot=events.length;
  if(!tot){
    document.getElementById('kpi-grid').innerHTML='<div class="kpi" style="grid-column:1/-1;text-align:center;color:var(--text-muted)">Nessun evento ancora. Aggiungi il primo dalla sezione "Nuovo evento".</div>';
    ['speaker-cards','chart-esiti','score-dist','score-dist-lbl','donut-imp-legend','stats-table','timeline'].forEach(id=>document.getElementById(id).innerHTML='');
    return;
  }
  const acc=events.filter(e=>e.esito==='Accettare').length;
  const dw=events.filter(e=>e.esito==='Accettare con downgrade').length;
  const rif=events.filter(e=>e.esito==='Rifiutare').length;
  const str=events.filter(e=>e.strategico==='Sì').length;
  const scores=events.map(e=>e.score);
  const avgScore=(scores.reduce((a,b)=>a+b,0)/scores.length).toFixed(1);
  const maxScore=Math.max(...scores), minScore=Math.min(...scores);
  const pctAcc=Math.round(acc/tot*100);

  document.getElementById('kpi-grid').innerHTML=[
    {l:'Totale eventi',v:tot,s:''},
    {l:'Accettare',v:acc,s:`${pctAcc}% del totale`},
    {l:'Con downgrade',v:dw,s:dw?`${Math.round(dw/tot*100)}%`:''},
    {l:'Rifiutati',v:rif,s:rif?`${Math.round(rif/tot*100)}%`:''},
    {l:'Strategici EF',v:str,s:`${Math.round(str/tot*100)}% del totale`},
    {l:'Score medio',v:avgScore,s:`min ${minScore} / max ${maxScore}`},
    {l:'Score più alto',v:maxScore,s:''},
    {l:'Score più basso',v:minScore,s:''},
  ].map(k=>`<div class="kpi"><div class="kpi-label">${k.l}</div><div class="kpi-val">${k.v}</div>${k.s?`<div class="kpi-sub">${k.s}</div>`:''}</div>`).join('');

  // Speaker cards
  const speakers=['Presidente','VP','DG','Villa','Lolli'];
  const spColors=['#3B6D11','#185FA5','#854F0B','#A32D2D','#533AB7'];
  const maxSpEv=Math.max(1,...speakers.map(sp=>events.filter(e=>e.speaker===sp).length));
  document.getElementById('speaker-cards').innerHTML=speakers.map((sp,i)=>{
    const cnt=events.filter(e=>e.speaker===sp).length;
    const spAcc=events.filter(e=>e.speaker===sp&&e.esito==='Accettare').length;
    const pct=Math.round(cnt/maxSpEv*100);
    return`<div class="sp-card">
      <div class="sp-avatar" style="background:${spColors[i]}22;color:${spColors[i]}">${sp.slice(0,2)}</div>
      <div class="sp-name">${sp}</div>
      <div class="sp-count">${cnt} event${cnt!==1?'i':'o'}</div>
      <div class="sp-bar"><div class="sp-bar-fill" style="width:${pct}%;background:${spColors[i]}"></div></div>
    </div>`;
  }).join('');

  // Esiti per speaker
  document.getElementById('chart-esiti').innerHTML=speakers.map(sp=>{
    const sAcc=events.filter(e=>e.speaker===sp&&e.esito==='Accettare').length;
    const sDw=events.filter(e=>e.speaker===sp&&e.esito==='Accettare con downgrade').length;
    const sRif=events.filter(e=>e.speaker===sp&&e.esito==='Rifiutare').length;
    const stot=sAcc+sDw+sRif; if(!stot) return '';
    const w=200;
    return`<div class="bar-row">
      <div class="bar-row-label">${sp}</div>
      <div class="bar-track">
        ${sAcc?`<div class="bar-seg" style="width:${Math.round(sAcc/stot*100)}%;background:var(--green-mid)">${sAcc}</div>`:''}
        ${sDw?`<div class="bar-seg" style="width:${Math.round(sDw/stot*100)}%;background:var(--amber-mid)">${sDw}</div>`:''}
        ${sRif?`<div class="bar-seg" style="width:${Math.round(sRif/stot*100)}%;background:var(--red-mid)">${sRif}</div>`:''}
      </div>
      <div class="bar-row-val">${stot}</div>
    </div>`;
  }).join('');

  // Score distribution
  const minS=Math.min(-3,...scores), maxS=Math.max(6,...scores);
  const buckets={};
  for(let i=minS;i<=maxS;i++) buckets[i]=0;
  scores.forEach(s=>{if(buckets[s]!==undefined)buckets[s]++;});
  const maxBucket=Math.max(1,...Object.values(buckets));
  document.getElementById('score-dist').innerHTML=Object.entries(buckets).map(([sc,cnt])=>`
    <div class="score-col">
      <div style="font-size:10px;color:var(--text-muted);min-height:14px">${cnt||''}</div>
      <div class="score-col-bar" style="height:${Math.max(4,Math.round(cnt/maxBucket*60))}px;background:${cnt?'var(--green-mid)':'var(--surface2)'}"></div>
    </div>`).join('');
  document.getElementById('score-dist-lbl').innerHTML=Object.keys(buckets).map(sc=>`<div style="flex:1;text-align:center">${sc}</div>`).join('');

  // Donut importanza
  const impCounts={Alta:0,Media:0,Bassa:0};
  events.forEach(e=>{if(impCounts[e.importanza]!==undefined)impCounts[e.importanza]++;});
  const impColors=['var(--red-mid)','var(--amber-mid)','var(--blue)'];
  const impLabels=Object.keys(impCounts);
  const canvas=document.getElementById('donut-imp');
  const ctx=canvas.getContext('2d');
  const total2=Object.values(impCounts).reduce((a,b)=>a+b,0)||1;
  let start=-Math.PI/2;
  ctx.clearRect(0,0,100,100);
  impLabels.forEach((k,i)=>{
    const slice=(impCounts[k]/total2)*2*Math.PI;
    ctx.beginPath(); ctx.moveTo(50,50);
    ctx.arc(50,50,40,start,start+slice);
    ctx.closePath();
    const colors=['#E24B4A','#EF9F27','#378ADD'];
    ctx.fillStyle=colors[i]; ctx.fill();
    start+=slice;
  });
  ctx.beginPath(); ctx.arc(50,50,22,0,2*Math.PI); ctx.fillStyle='#fff'; ctx.fill();
  document.getElementById('donut-imp-legend').innerHTML=impLabels.map((k,i)=>{
    const pct2=Math.round(impCounts[k]/total2*100);
    const colors=['#E24B4A','#EF9F27','#378ADD'];
    return`<div class="legend-item"><div class="legend-dot" style="background:${colors[i]}"></div>${k}: ${impCounts[k]} (${pct2}%)</div>`;
  }).join('');

  // Stats table
  const agendaOk=events.filter(e=>e.agenda==='Sì').length;
  const caricoAlto=events.filter(e=>e.carico==='Alto').length;
  const impAlta=events.filter(e=>e.importanza==='Alta').length;
  document.getElementById('stats-table').innerHTML=[
    ['Agenda compatibile',`${agendaOk} / ${tot} (${Math.round(agendaOk/tot*100)}%)`],
    ['Carico alto al momento evento',`${caricoAlto} / ${tot} (${Math.round(caricoAlto/tot*100)}%)`],
    ['Importanza Alta',`${impAlta} / ${tot} (${Math.round(impAlta/tot*100)}%)`],
    ['Tasso accettazione diretta',`${acc} / ${tot} (${pctAcc}%)`],
    ['Tasso downgrade',`${dw} / ${tot} (${Math.round(dw/tot*100)}%)`],
    ['Tasso rifiuto',`${rif} / ${tot} (${Math.round(rif/tot*100)}%)`],
  ].map(([l,val])=>`<tr><td class="td-label">${l}</td><td class="td-val">${val}</td></tr>`).join('');

  // Timeline
  const last10=[...events].sort((a,b)=>(b.ts||0)-(a.ts||0)).slice(0,10);
  document.getElementById('timeline').innerHTML=last10.map(e=>{
    const s=esitoStyle(e.esito);
    return`<div class="tl-item">
      <div class="tl-date">${fmtDate(e.data)}</div>
      <div class="tl-dot" style="background:${s.bar}"></div>
      <div class="tl-content">
        <div class="tl-name">${e.nome||'—'}</div>
        <div class="tl-meta">${e.speaker} · Score ${e.score} · <span style="color:${s.color}">${e.esito}</span>${e.luogo?' · '+e.luogo:''}</div>
      </div>
    </div>`;
  }).join('');
}

function goTo(id) {
  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
  document.querySelectorAll('.nav-item').forEach(n=>n.classList.remove('active'));
  document.getElementById('page-'+id).classList.add('active');
  event.currentTarget.classList.add('active');
  if(id==='archivio') renderArchivio();
  if(id==='dashboard') renderDashboard();
}
